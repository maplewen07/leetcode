using namespace std;

typedef int(* FunPtr)(int);

void f1(int nums[], int len, FunPtr fp) {
	for (int i = 0; i < len; i++) {
		nums[i] = fp(nums[i]);
	}
}

int f2 (int num) {
	return num * num;
}

int main() {
