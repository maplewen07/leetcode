class Solution {
public:
    int findMaxConsecutiveOnes(vector<int>& nums) {
		int ans = 0;
        int cnt = 0, len = nums.size();
		
		for (int i = 0; i < len; i++) {
			if (nums[i] == 1) {
				cnt++;
			} else {
				ans = max(cnt, ans);
				cnt = 0;
			}
		}
		ans = max(cnt, ans);
		return ans;
    }
};
