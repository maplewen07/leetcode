/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        if (head == nullptr || head->next == nullptr) {
            return true;
        }

        ListNode *slow = head, *fast = head;
        while (fast != nullptr && fast->next != nullptr) {
            slow = slow->next;
            fast = fast->next->next;
        }
        // slow 为中间，或者第二段开头第一个
        // fast == nullptr 第二段开头， fast != nullptr 为中间。

        ListNode *pre = nullptr, *ptr = slow, *next = slow->next;
        while (next != nullptr) {
            // cout << ptr->val << endl;
            pre = ptr;
            ptr = next;
            next = next->next;

            ptr->next = pre;
        }
        slow->next = nullptr;
        ListNode* cur = head;
        while (ptr != nullptr) {
            if (ptr->val != cur->val) {
                return false;
            }
            ptr = ptr->next;
            cur = cur->next;
        }

        return true;
    }
};


class Solution {
public:
    bool isPalindrome(ListNode* head) {
        if (head == nullptr || head->next == nullptr) {
            return true;
        }

        ListNode* middle = findMiddleNode(head);
        ListNode* tail = reverseNode(middle->next);

        bool ans = true;
        ListNode *ptr = head, *cur = tail;
        while (cur != nullptr) {
            if (ptr->val != cur->val) {
                ans = false;
            }
            ptr = ptr->next;
            cur = cur->next;
        }

        middle->next = reverseNode(tail);
        return ans;
    }

private:
    ListNode* findMiddleNode(ListNode* head) {
        ListNode* slow = head;
        ListNode* fast = head->next;

        while (fast != nullptr && fast->next != nullptr) {
            slow = slow->next;
            fast = fast->next->next;
        }
        return slow;
        // 当 fast != nullptr, slow 为第一个中点 len % 2 == 0;
        // 当 fast == nullptr, slow 为中间。 len % 2 == 1;
    }

    ListNode* reverseNode(ListNode* head) {
        ListNode* prev = nullptr;
        ListNode* curr = head;
        while (curr != nullptr) {
            ListNode* tmp = curr->next;
            curr->next = prev;

            prev = curr;
            curr = tmp;
        }
        return prev;
    }
};
